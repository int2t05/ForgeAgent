## 2. 任务执行 (Task Execution)

### 2.1 创建并启动任务
```python
POST /api/v1/tasks
{
    session_id: str,
    message: str,           # 用户消息

# 伪代码
async def create_task(session_id: str, message: str):
    # 1. 验证会话存在
    session = await db.get(Session, session_id)
    if not session:
        raise HTTPException(404, "Session not found")

    # 2. 存储用户消息
    await db.add(Message(
        session_id=session_id,
        role="user",
        content=message,
        created_at=now()
    ))
    await db.commit()

    # 3. 创建任务
    task_id = uuid4()
    task = Task(
        id=task_id,
        session_id=session_id,
        status="running",
        plan_version=0,
        created_at=now()
    )
    await db.add(task)
    await db.commit()

    # 4. 异步执行 Agent
    asyncio.create_task(run_agent_task(task_id, message))

    return TaskResponse(
        id=task_id,
        status="running",
        events_stream_path=f"/tasks/{task_id}/events/stream"
    )
```

### 2.2 LangGraph 执行流程
```python
# 伪代码: run_agent_task(task_id, message)

async def run_agent_task(task_id: str, message: str):
    state = AgentState(
        task_id=task_id,
        messages=[message],
        plan=None,
        plan_version=0,
        outcome=None
    )

    for attempt in range(max_replan_attempts):
        # 执行图
        state = await graph.ainvoke(state)

        # 检查是否需要重规划
        if state.get("needs_replan"):
            state["plan_version"] += 1
            await write_event(task_id, "replan_record", {"version": state["plan_version"]})
            continue

        # 检查执行结果
        if state.get("outcome") in ("success", "failed"):
            break

    # 更新任务状态
    await db.update(Task).where(Task.id == task_id).values(
        status=state.get("outcome", "failed"),
        summary=state.get("summary"),
        updated_at=now()
    )
```

### 2.3 任务节点 (Planner & Executor)
```python
# Planner Node - 创建执行计划
def planner_node(state: AgentState) -> AgentState:
    plan = generate_plan(state["messages"])
    return state.copy(
        plan={"steps": plan},
        needs_replan=False
    )

# Executor Node - 执行计划步骤
def executor_node(state: AgentState) -> AgentState:
    plan = state["plan"]
    for step in plan["steps"]:
        # 记录步骤开始事件
        write_event(state["task_id"], "step_start", step)
        result = execute_tool(step["tool"], step["args"])
        state["messages"].append(result)

        if step.get("fail"):
            return state.copy(outcome="failed")
    return state.copy(outcome="success")
```

## 3. SSE 事件流 (Event Streaming)

### 3.1 事件流端点
```python
GET /api/v1/tasks/{task_id}/events/stream?after_seq=0

# 伪代码
async def stream_task_events(task_id: str, after_seq: int = 0):
    last_seq = after_seq
    empty_rounds = 0

    while True:
        events = await db.query(TaskEvent)
            .filter(TaskEvent.task_id == task_id)
            .filter(TaskEvent.seq > last_seq)
            .order_by(TaskEvent.seq)
            .all()

        if events:
            empty_rounds = 0
            for event in events:
                yield f"event: {event.kind}\ndata: {event.payload_json}\n\n"
                last_seq = event.seq
        else:
            empty_rounds += 1

        # 终端状态4轮轮询后关闭
        task = await db.get(Task, task_id)
        if task.status in ("success", "failed", "cancelled") and empty_rounds >= 4:
            break

        await asyncio.sleep(0.15)  # 轮询间隔
```

### 3.2 写入任务事件

```python
# 伪代码
async def write_event(task_id: str, kind: str, payload: dict):
    seq = await db.execute(
        select(func.count()).where(TaskEvent.task_id == task_id)
    ) + 1

    event = TaskEvent(
        task_id=task_id,
        seq=seq,
        ts=now(),
        module=payload.get("module", "execution"),
        kind=kind,
        payload_json=json.dumps(payload)
    )
    await db.add(event)
    await db.commit()
    return event
```

## 4. 工具注册 (Tool Registry)

### 4.1 获取所有工具
```python
GET /api/v1/tools

# 伪代码
def get_all_tools() -> List[dict]:
    tools = []

    # 内置工具
    tools.extend([
        {"name": "echo", "description": "...", "module": "builtin"},
        {"name": "mock_search", "description": "...", "module": "builtin"}
    ])

    # MCP 工具 (从设置加载)
    mcp_config = get_settings("mcp")
    for server_name, tools_list in mcp_config.get("tools", {}).items():
        for tool in tools_list:
            tools.append({**tool, "module": "mcp", "server": server_name})

    # Skill 工具 (从配置的路径加载)
    skills_paths = get_settings("skills_paths", [])
    for path in skills_paths:
        tools.extend(load_skill_tools(path))

    return tools
```

### 4.2 工具执行
```python
# 伪代码
async def execute_tool(name: str, args: dict) -> dict:
    tool = registry.get(name)
    if not tool:
        return {"error": f"Tool '{name}' not found"}

    if tool["module"] == "builtin":
        return await builtin_tools[name](args)
    elif tool["module"] == "mcp":
        return await mcp_tools[name](args)
    elif tool["module"] == "skill":
        return await skill_tools[name](args)
```

```python
    for key, value in settings.items():
        # 安全检查: 禁止包含敏感关键词
```
